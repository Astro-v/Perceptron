# Set the minimum required version of CMake
cmake_minimum_required(VERSION 3.5)

# Set the project name
project(PerceptronTests)

# Find the GoogleTest library
find_package(GTest REQUIRED)

# Add the test executable
add_executable(PerceptronTests test_perceptron.cpp)

# Link the test executable with the Perceptron library and the GoogleTest library
target_link_libraries(PerceptronTests PRIVATE Perceptron GTest::GTest GTest::Main)

# Add the test to CTest
add_test(NAME PerceptronTests COMMAND PerceptronTests)
